    CREATE DATABASE VandorBakancsDb;

-- RÉGI TÁBLÁK TÖRLÉSE 
IF OBJECT_ID('KedvencTurak', 'U') IS NOT NULL DROP TABLE KedvencTurak;
IF OBJECT_ID('ForumKommentek', 'U') IS NOT NULL DROP TABLE ForumKommentek;
IF OBJECT_ID('ForumTemak', 'U') IS NOT NULL DROP TABLE ForumTemak;
IF OBJECT_ID('Turak', 'U') IS NOT NULL DROP TABLE Turak;
IF OBJECT_ID('Users', 'U') IS NOT NULL DROP TABLE Users;
GO

-- FELHASZNÁLÓK TÁBLA
CREATE TABLE Users (
    id int IDENTITY(1,1) PRIMARY KEY,
    nev nvarchar(100) NOT NULL,
    email nvarchar(100) UNIQUE NOT NULL,
    password nvarchar(255) NOT NULL,
    role nvarchar(20) DEFAULT 'user'
);

-- TÚRÁK TÁBLA
CREATE TABLE Turak (
    id int IDENTITY(1,1) PRIMARY KEY,
    nev nvarchar(255) NOT NULL,
    helyszin nvarchar(255) NOT NULL,
    idotartam nvarchar(50) NOT NULL,
    nehezseg nvarchar(50) NOT NULL,
    leiras nvarchar(MAX) NULL,
    kinek_ajanljuk nvarchar(MAX) NULL
);

-- FÓRUM TÉMÁK TÁBLA
CREATE TABLE ForumTemak (
    id int IDENTITY(1,1) PRIMARY KEY,
    cim nvarchar(255) NOT NULL,
    szerzo nvarchar(100) NOT NULL,
    datum datetime DEFAULT GETDATE()
);

-- FÓRUM KOMMENTEK TÁBLA
CREATE TABLE ForumKommentek (
    id int IDENTITY(1,1) PRIMARY KEY,
    tema_id int NOT NULL,
    szerzo nvarchar(100) NOT NULL,
    szoveg nvarchar(MAX) NOT NULL,
    datum datetime DEFAULT GETDATE(),
    CONSTRAINT FK_Komment_Tema FOREIGN KEY (tema_id) REFERENCES ForumTemak(id) ON DELETE CASCADE
);

-- KEDVENC TÚRÁK TÁBLA 
CREATE TABLE KedvencTurak (
    id INT IDENTITY(1,1) PRIMARY KEY,
    user_id INT NOT NULL,
    tura_id INT NOT NULL,
    datum DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (user_id) REFERENCES Users(id) ON DELETE CASCADE,
    FOREIGN KEY (tura_id) REFERENCES Turak(id) ON DELETE CASCADE,
    CONSTRAINT UC_UserTura UNIQUE (user_id, tura_id) -- ne lehessen kétszer szívezni ugyanazt
);
